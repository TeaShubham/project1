package com.example.timer;

import android.os.Bundle;
import android.os.CountDownTimer;
import android.view.View;
import android.widget.Button;
import android.widget.TextView;
import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {

    private CountDownTimer countDownTimer;
    private long timeLeftInMillis = 60000; // 60 seconds
    private boolean timerRunning = false;

    private TextView elapsedTimeView;
    private TextView remainingTimeView;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        Button startButton = findViewById(R.id.start_button);
        elapsedTimeView = findViewById(R.id.elapsed_time);
        remainingTimeView = findViewById(R.id.remaining_time);

        startButton.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                startTimer();
            }
        });
    }

    private void startTimer() {
        if (timerRunning) return;

        countDownTimer = new CountDownTimer(timeLeftInMillis, 1000) {
            private long elapsedTime = 0;

            @Override
            public void onTick(long millisUntilFinished) {
                elapsedTime += 1000;
                timeLeftInMillis = millisUntilFinished;
                updateUI(elapsedTime, millisUntilFinished);
            }

            @Override
            public void onFinish() {
                timerRunning = false;
                updateUI(elapsedTime, 0);
            }
        }.start();

        timerRunning = true;
    }

    private void updateUI(long elapsedTime, long millisUntilFinished) {
        elapsedTimeView.setText("Elapsed Time: " + (elapsedTime / 1000) + "s");
        remainingTimeView.setText("Remaining Time: " + (millisUntilFinished / 1000) + "s");
    }
}
